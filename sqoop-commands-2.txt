sqoop list-databases --connect jdbc:mysql://localhost:3306 --username root 

sqoop list-tables --connect jdbc:mysql://localhost:3306/ctsdb --username root

sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table users --num-mappers 1 --target-dir /user/cloudera/sqoop/users --fields-terminated-by ',' --lines-terminated-by '\n' 

-- import data in overwrite mode --
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table users --num-mappers 1 --target-dir /user/cloudera/sqoop/users --fields-terminated-by ',' --lines-terminated-by '\n' --delete-target-dir

-- import data in append mode --
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table users --num-mappers 1 --target-dir /user/cloudera/sqoop/users --fields-terminated-by ',' --lines-terminated-by '\n' --append

-- import data with enclosing char --
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table users -m 1 --target-dir /user/cloudera/sqoop/users4 --enclosed-by '"' --delete-target-dir

-- import data in avro format --
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table users -m 1 --target-dir /user/cloudera/sqoop/users-avro --delete-target-dir -as-avrodatafile

-- import few columns of a table --
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table students --columns "id,name,phone,email" --num-mappers 1 --target-dir /user/cloudera/sqoop/students2

-- import based on a query --
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --query "select country, count(1) as count from students where 1=1 AND \$CONDITIONS group by country" --num-mappers 1 --target-dir /user/cloudera/sqoop/students3

-- import using split-by --
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table students -m 3 --split-by 'id'  --target-dir /user/cloudera/sqoop/students1 --fields-terminated-by ','

====== Working with Hive =====

-- create a hive table as per the schema of a mysql table ---
sqoop create-hive-table --connect jdbc:mysql://localhost:3306/ctsdb --username root --table users --fields-terminated-by ',' --hive-table sqoopdb.students

-- import data into a hive table from a mysql table ---
sqoop import --connect jdbc:mysql://localhost:3306/ctsdb --username root --table students --hive-table sqoopdb.students --fields-terminated-by ',' -m 1 --hive-import 


=======  Evaluating MySQL from Linux using Sqoop =========

sqoop eval --connect jdbc:mysql://localhost:3306/ctsdb --username root --query 'select * from students'

sqoop eval --connect jdbc:mysql://localhost:3306/ctsdb --username root --query 'create table tab1(col1 INT, col2 VARCHAR(50))'

sqoop eval --connect jdbc:mysql://localhost:3306/ctsdb --username root --query 'insert into tab1 values (1, "raju"), (2, "ramesh")'

sqoop eval --connect jdbc:mysql://localhost:3306/ctsdb --username root --query 'update tab1 set col2="kanakaraju" where col1=1'

sqoop eval --connect jdbc:mysql://localhost:3306/ctsdb --username root --query 'delete from tab1 where col1=2'

sqoop eval --connect jdbc:mysql://localhost:3306/ctsdb --username root --query 'drop table tab1'
